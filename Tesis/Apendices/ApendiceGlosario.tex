\newglossaryentry{DDM}
{
    name={Distribución Multinomial},
    text={distribución de datos multinomial},
    description={La distribución multinomial es una generalización de la distribución binomial. La misma permite que pueda haber múltiples resultados, en lugar de solo dos posibles, para cada ensayo. La función de probabilidad de la distribución multinomial es:
    \begin{equation} \begin{aligned}
f(x_1\ldots x_k;n,p_1\ldots p_k) & {} = \begin{cases} { \displaystyle {n! \over x_1!\cdots x_k!}p_1^{x_1}\cdots p_k^{x_k}} \quad &
\mbox{cuando} \sum_{i=1}^k x_i=n \\  \\
0 & \mbox{en otros casos} \end{cases}
\end{aligned}
\end{equation}}
}
